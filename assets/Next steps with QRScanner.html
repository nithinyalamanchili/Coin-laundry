Future<void> _applyCoupon(String code) async {
  if (_franchiseId == null) {
    _showSnack("Franchise info missing");
    return;
  }

  final prefs = await SharedPreferences.getInstance();
  final token = prefs.getString('auth_token');
  if (token == null) {
    _showSnack("Please log in again");
    return;
  }

  try {
    final resp = await http.post(
      Uri.parse('https://api.coinlaundryindia.com/promocode'),
      headers: {
        'Authorization': 'Bearer $token',
        'Content-Type': 'application/json',
      },
      body: json.encode({
        "coupon": code,
        "franchiseId": _franchiseId,
      }),
    );

    if (resp.statusCode == 200) {
      final data = json.decode(resp.body);
      final discount = (data['coins'] ?? 0).toInt();
      setState(() {
        _payableInr = (_payableInr - discount).clamp(0, _payableInr);
      });
      _showSnack("Coupon applied: -$discount INR");
    } else {
      _showSnack("Invalid or expired coupon");
    }
  } catch (e) {
    _showSnack("Failed to apply coupon");
  }
}
